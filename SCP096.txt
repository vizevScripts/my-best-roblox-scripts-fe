local Notification = loadstring(game:HttpGet("https://raw.githubusercontent.com/vizevScripts/my-best-roblox-scripts-fe/refs/heads/main/Notification%20Module.lua"))()
local function CreateNotification(Message, SoundType)
	if SoundType == "W" then
		Notification:SetNotificationSound("Warning")
	elseif SoundType == "N" then
		Notification:SetNotificationSound("Notification")
	elseif not SoundType or SoundType and SoundType ~= "W" or SoundType ~= "N" then
		SoundType = "N"
	end
	Notification:CreateNotification("From SCP-096 script: ", Message, 5)
end

CreateNotification("Hi there! Please wait for SCP-096 FE script!", "N")

local Players = game:GetService("Players")
local Scp096Player = Players.LocalPlayer
local Scp096Backpack = Scp096Player:FindFirstChildOfClass("Backpack")
local Scp096Character = Scp096Player.Character or Scp096Player.CharacterAdded:Wait()
local Scp096Humanoid = Scp096Character:FindFirstChildOfClass("Humanoid")
local Scp096HRP = Scp096Character:FindFirstChild("HumanoidRootPart")

local OldScp096Walkspeed = Scp096Humanoid.WalkSpeed
local OldScp096JumpPower = Scp096Humanoid.JumpPower

local Scp096CryAnim = Instance.new("Animation")
Scp096CryAnim.AnimationId = "rbxassetid://139672038778026"

Scp096Cry.Looped = true

Scp096PanicAnim.AnimationId = "rbxassetid://132065685005919"
local Scp096Panic = Scp096Humanoid:LoadAnimation(Scp096PanicAnim)
Scp096Panic.Looped = true

local ReadyToAttack = false

ScpCryTool.CanBeDropped = false

	if Scp096Panic.IsPlaying then
		Scp096Panic:Stop()
	end
	if not Scp096Cry.IsPlaying then
		Scp096Cry:Play()
	end
	if ReadyToAttack then
		ReadyToAttack = false
	end
	Scp096Humanoid.WalkSpeed = 0
	Scp096Humanoid.JumpPower = 0
end)

local function StartToFling()
	for _, NPC in pairs(workspace:GetChildren()) do
		if NPC and NPC.Name ~= Scp096Character.Name and NPC:FindFirstChild("HumanoidRootPart") or NPC:FindFirstChild("Head") and NPC:FindFirstChildOfClass("Humanoid") and NPC:FindFirstChildOfClass("Humanoid").Health > 0 then
			local NPCHumanoid = NPC:FindFirstChildOfClass("Humanoid")
			local NPCPart = NPC:FindFirstChild("HumanoidRootPart") or NPC:FindFirstChild("Head")
			if (Scp096HRP.Position - NPCPart.Position).Magnitude < 15 then
				if not Scp096HRP:FindFirstChild("BodyThrust") then
					local BodyThrust = Instance.new("BodyThrust")
					BodyThrust.Name = "BodyThrust"
					BodyThrust.Force = Vector3.new(999, 999, 999)
					repeat wait()
						BodyThrust.Location = NPCPart.Position
						Scp096HRP.CFrame = CFrame.new(Scp096HRP.Position, NPCPart.Position)
						Scp096HRP.Velocity = Scp096HRP.CFrame.LookVector * 9999
					until NPCHumanoid.Health == 0 and not NPCPart
					BodyThrust:Destroy()
					ReadyToAttack = false
				end
			end
		end
	end
end

while wait() do
	if Scp096Player and Scp096Character and Scp096Humanoid and Scp096Humanoid.Health > 0 and Scp096HRP then
		if ReadyToAttack then
			Scp096Humanoid.WalkSpeed = 85
			Scp096Humanoid.JumpPower = 180
			StartToFling()
		end
		if Scp096Humanoid.RigType == Enum.HumanoidRigType.R6 then
			if ReadyToAttack then
				ReadyToAttack = false
			end
			if Scp096Character:FindFirstChild("scp096 cry") or Scp096Backpack:FindFirstChild("scp096 cry") then
				local SCPAnimTool = Scp096Character:FindFirstChild("scp096 cry") or Scp096Backpack:FindFirstChild("scp096 cry")
				SCPAnimTool:Destroy()
			end
			if Scp096Character:FindFirstChild("Scp-096 Panic") or Scp096Backpack:FindFirstChild("Scp-096 Panic") then
				local SCPAnimTool = Scp096Character:FindFirstChild("Scp-096 Panic") or Scp096Backpack:FindFirstChild("Scp-096 Panic")
				SCPAnimTool:Destroy()
			end
			if Scp096Character:FindFirstChild("SCP-096 Run Anim") or Scp096Backpack:FindFirstChild("SCP-096 Run Anim") then
				local SCPAnimTool = Scp096Character:FindFirstChild("SCP-096 Run Anim") or Scp096Backpack:FindFirstChild("SCP-096 Run Anim")
				SCPAnimTool:Destroy()
			end
			CreateNotification("Your character is R6 rig type. Use R15 rig type!", "W")
			Scp096Humanoid.Health = 0
			break
		end
	end
end
