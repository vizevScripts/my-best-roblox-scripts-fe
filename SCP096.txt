local Players = game:GetService("Players")
local Scp096Player = Players.LocalPlayer
local Scp096Backpack = Scp096Player:FindFirstChildOfClass("Backpack")
local Scp096Character = Scp096Player.Character or Scp096Player.CharacterAdded:Wait()
local Scp096Humanoid = Scp096Character:FindFirstChildOfClass("Humanoid")
local Scp096HRP = Scp096Character:FindFirstChild("HumanoidRootPart")

local OldScp096Walkspeed = Scp096Humanoid.WalkSpeed
local OldScp096JumpPower = Scp096Humanoid.JumpPower

local Scp096CryAnim = Instance.new("Animation")
Scp096CryAnim.AnimationId = "rbxassetid://139672038778026"
local Scp096Cry = Scp096Humanoid:LoadAnimation(Scp096CryAnim)
Scp096Cry.Looped = true
Scp096Cry:AdjustSpeed(.3)

local Scp096PanicAnim = Instance.new("Animation")
Scp096PanicAnim.AnimationId = "rbxassetid://132065685005919"
local Scp096Panic = Scp096Humanoid:LoadAnimation(Scp096PanicAnim)
Scp096Panic.Looped = true
Scp096Panic:AdjustSpeed(10000000000)

local ReadyToAttack = false

local ScpCryTool = Instance.new("Tool", Scp096Backpack)
ScpCryTool.Name = "scp096 cry"
ScpCryTool.ToolTip = "SCP-096 Crying"
ScpCryTool.RequiresHandle = false
ScpCryTool.CanBeDropped = false

ScpCryTool.Equipped:Connect(function()
	if Scp096Panic.IsPlaying then
		Scp096Panic:Stop()
	end
	if not Scp096Cry.IsPlaying then
		Scp096Cry:Play()
	end
	if ReadyToAttack then
		ReadyToAttack = false
	end
	Scp096Humanoid.WalkSpeed = 0
	Scp096Humanoid.JumpPower = 0
end)

ScpCryTool.Unequipped:Connect(function()
	if Scp096Cry.IsPlaying then
		Scp096Cry:Stop()
	end
	Scp096Humanoid.WalkSpeed = OldScp096Walkspeed
	Scp096Humanoid.JumpPower = OldScp096JumpPower
end)

local ScpPanicTool = Instance.new("Tool", Scp096Backpack)
ScpPanicTool.Name = "Scp-096 Panic"
ScpPanicTool.ToolTip = "SCP-096 Panic"
ScpPanicTool.RequiresHandle = false
ScpPanicTool.CanBeDropped = false

ScpPanicTool.Equipped:Connect(function()

end)

local ScpRunTool = Instance.new("Tool", Scp096Backpack)
ScpRunTool.Name = "Scp-096 Panic"
ScpRunTool.ToolTip = "SCP-096 Panic"
ScpRunTool.RequiresHandle = false
ScpRunTool.CanBeDropped = false

ScpRunTool.Equipped:Connect(function()
	if Scp096Panic.IsPlaying then
		Scp096Panic:Stop()
	end
	if Scp096Cry.IsPlaying then
		Scp096Cry:Stop()
	end
	ReadyToAttack = true
end)

ScpRunTool.Unequipped:Connect(function()
	if ReadyToAttack then
		ReadyToAttack = false
	end
	Scp096Humanoid.WalkSpeed = OldScp096Walkspeed
	Scp096Humanoid.JumpPower = OldScp096JumpPower
end)

local function StartToFling()
	for _, NPC in pairs(workspace:GetChildren()) do
		if NPC and NPC.Name ~= Scp096Character.Name and NPC:FindFirstChild("HumanoidRootPart") or NPC:FindFirstChild("Head") and NPC:FindFirstChildOfClass("Humanoid") and NPC:FindFirstChildOfClass("Humanoid").Health > 0 then
			local NPCHumanoid = NPC:FindFirstChildOfClass("Humanoid")
			local NPCPart = NPC:FindFirstChild("HumanoidRootPart") or NPC:FindFirstChild("Head")
			if (Scp096HRP.Position - NPCPart.Position).Magnitude < 15 then
				if not Scp096HRP:FindFirstChild("Thrust") then
					local Thrust = Instance.new("BodyThrust", Scp096HRP)
					Thrust.Name = "Thrust"
					Thrust.Force = Vector3.new(math.huge, math.huge, math.huge)
					repeat wait()
						Thrust.Location = NPCPart.Position
						Scp096HRP.CFrame = NPCPart.CFrame
					until NPCHumanoid.Health == 0
					Thrust:Destroy()
					ReadyToAttack = false
				end
			end
		end
	end
end

while wait() do
	if Scp096Player and Scp096Character and Scp096Humanoid and Scp096Humanoid.Health > 0 and Scp096HRP then
		if ReadyToAttack then
			Scp096Humanoid.WalkSpeed = 50
			StartToFling()
		end
	end
end
